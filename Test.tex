\question[10] Evaluate and simplify the difference quotient, $\dfrac{f(x+h)-f(x)}{h}$, for the function.\\ \\
\(f(x) = 5 \, x^{2} - 2\)

%
\begin{solution}
\(f(x+h) = 5(x+h)^2 - 2\) 

 \(= 5(x^2 + 2xh + h^2) - 2\) 

 \(= 5 \, h^{2} + 10 \, h x + 5 \, x^{2} - 2\) 

 \(\displaystyle{\frac{f(x+h)-f(x)}{h} = \frac{ (5 \, h^{2} + 10 \, h x + 5 \, x^{2} - 2) - (5 \, x^{2} - 2) }{h}}\) 

 \(=\displaystyle{ \frac{ 5 \, h^{2} + 10 \, h x }{h}}\) 

 \(= \displaystyle{ \frac{ h(5 \, h + 10 \, x) }{h}}\) 

 \(\boxed{ 5 \, h + 10 \, x }\)
\end{solution}
\vspace{\stretch{1}}\answerline

\question[3]
An architect is designing a custom triangular stained-glass window for a modern building. The three sides of the glass pane will measure \(35\) inches, \(77\) inches, and \(46\) inches. What is the area of the glass required for this pane?

%
\begin{solution}
\(s = \frac{a + b + c}{2}\)

\(s = \frac{35 + 77 + 46}{2} = 79.0000000000000\)

\(\text{Area} = \sqrt{s(s-a)(s-b)(s-c)}\)

\(\text{Area} = \sqrt{79.0000000000000(79.0000000000000-35)(79.0000000000000-77)(79.0000000000000-46)}\)

\(\text{Area} = \sqrt{79.0000000000000(44.0000000000000)(2.00000000000000)(33.0000000000000)}\)

\(\boxed{\text{Area} \approx 478.97}\)
\end{solution}
\vspace{\stretch{1}}
\answerline

   \part \(\displaystyle \lim_{x \to 1} \frac{x^{2} - 4 \, x + 3}{x^{2} - 1}\)
\vspace{\stretch{1}}
\answerline
   \part \(\displaystyle \lim_{x \to 6} \frac{x^{2} - 3 \, x - 4}{x + 1}\)
\vspace{\stretch{1}}
\answerline
   \part \(\displaystyle \lim_{x \to 7} \sqrt{x^{2} + 4 \, x + 4}\)

\(\displaystyle \lim_{x \to 6} \frac{x^{2} - 3 \, x - 4}{x + 1} = \frac{14}{7} = 2\)

\(\displaystyle \lim_{x \to 7} \sqrt{x^{2} + 4 \, x + 4} = \sqrt{81} = 9\)

\(\displaystyle \cos\left(2\cos^{-1}\left(\frac{5u}{13}\right)\right)\)

\(\boxed{\frac{50u^2 - 169}{169}}\)
\end{solution}

    \answerline

\question[3]
Solve the following trigonometric equation on the interval \([0, 2\pi)\).\\ \\
\(2\sin^2 \theta + 7\sin \theta = -3\)

%
\begin{solution}
\(2\sin^2 \theta + 7\sin \theta + 3 = 0\)

\((\sin \theta + 1)(\sin \theta + 3) = 0\)

\(\sin \theta + 1 = 0 \quad \text{OR} \quad \sin \theta + 3 = 0\)

\(\sin \theta = -1 \quad \text{OR} \quad \sin \theta = -3\)

\(\sin \theta = -1 \quad \text{OR} \quad \text{No real solution}\)

\(\boxed{\theta = \frac{7\pi}{6}, \frac{11\pi}{6}}\)
\end{solution}
    \vspace{\stretch{1}}
    \answerline
